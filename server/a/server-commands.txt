# Команды для настройки Xpanel на сервере 185.250.180.254

## 1. Подключение к серверу
ssh root@185.250.180.254

## 2. Распаковка проекта
cd /root
unzip -o a.zip
cd Xpanels

## 3. Запуск установки
chmod +x setup-server.sh
./setup-server.sh

## 4. Настройка email (после установки)
nano .env
# Измените строки:
# EMAIL_USER=your-email@gmail.com
# EMAIL_PASS=your-gmail-app-password

## 5. Перезапуск приложения
pm2 restart xpanel

## 6. Проверка статуса
pm2 status
pm2 logs xpanel

## 7. Проверка Nginx
systemctl status nginx
nginx -t

## 8. Проверка SSL сертификата
certbot certificates

## Полезные команды:
- pm2 restart xpanel    # Перезапуск приложения
- pm2 logs xpanel       # Просмотр логов
- pm2 monit            # Мониторинг ресурсов
- systemctl reload nginx # Перезагрузка Nginx
- /root/backup-xpanel.sh # Создание бэкапа
