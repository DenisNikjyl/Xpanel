# Быстрые команды для настройки SSL на сервере

## 1. Подключение к серверу
ssh root@185.250.180.254

## 2. Проверка DNS (должен показать IP сервера)
dig +short xpanel.xload.ru

## 3. Основная установка (если еще не выполнена)
cd /root
unzip -o a.zip
cd a
chmod +x setup-server.sh
./setup-server.sh

## 4. Если SSL не установился автоматически
chmod +x ssl-setup-manual.sh
./ssl-setup-manual.sh

## 5. Ручная настройка SSL (если нужно)
apt install -y certbot python3-certbot-nginx
certbot --nginx -d xpanel.xload.ru --email fillsites0@gmail.com --agree-tos --redirect

## 6. Проверка статуса
systemctl status nginx
pm2 status
certbot certificates

## 7. Открытие портов (если нужно)
ufw allow 80/tcp
ufw allow 443/tcp
ufw reload

## Результат:
# HTTP: http://xpanel.xload.ru (перенаправляется на HTTPS)
# HTTPS: https://xpanel.xload.ru (основной доступ)

## Архитектура:
# Пользователь -> HTTPS (443) -> Nginx -> Node.js (3001)
# SSL сертификат обрабатывается Nginx
# Node.js работает на внутреннем порту 3001
